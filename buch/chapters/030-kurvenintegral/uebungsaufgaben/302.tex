\begin{align*}
\oint_{\gamma} \alpha
&=
\int_0^{2\pi} \langle \alpha(\gamma(t)),\dot{\gamma}(t)\rangle \,dt

\int_0^{2\pi}

\int_0^{2\pi}
\sin t\,\langle dx,\dot{\gamma}(t)\rangle

\cos t\,\langle dy,\dot{\gamma}(t)\rangle

\int_0^{2\pi}
-
\sin^2 t
-
\cos^2 t
\,dt
=
-\int_0^{2\pi}\,dt

\end{align*}
\end{loesung}
