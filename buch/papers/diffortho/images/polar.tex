\documentclass[tikz]{standalone}
\usepackage{amsmath}
\usepackage{times}
\usepackage{txfonts}
\usepackage{pgfplots}
\usepackage{csvsimple}
\usetikzlibrary{arrows,intersections,math,calc}
\begin{document}
\def\skala{1}
\definecolor{darkred}{rgb}{0.8,0,0}
\begin{tikzpicture}[>=latex,thick,scale=\skala]
\clip (-3.1,-3.1) rectangle (3.4,3.5);

\foreach \x in {-3,-2.5,...,3}{
	\draw[color=gray!40] (\x,-3) -- (\x,3);
	\draw[color=gray!40] (-3,\x) -- (3,\x);
}

\def\winkelradius{0.10}
\begin{scope}
\clip (-3.1,-3.1) rectangle (3.1,3.1);
\foreach \w in {0,15,...,350}{
	\foreach \r in {0.5,1,...,4}{
		\fill[color=blue!40] (\w:\r) -- ++(\w:\winkelradius)
			arc(\w:{\w+90}:\winkelradius) -- cycle;
	}
}

\foreach \r in {0.5,1,...,4}{
	\draw[color=darkred!60] (0,0) circle[radius=\r];
}
\foreach \w in {0,15,...,350}{
	\draw[color=darkred!60] (0,0) -- (\w:5.1);
}
\end{scope}

\draw[->,color=black] (-3.1,0) -- (3.3,0) coordinate[label={$y^1$}];
\draw[->,color=black] (0,-3.1) -- (0,3.3) coordinate[label={right:$y^2$}];

\draw[->,line width=1.4pt,color=darkred] (0,0) -- (3.3,0);
\node[color=darkred] at (2.75,0.05) [below] {$x^1$};
\draw[line width=1.4pt,color=darkred] (2,0) arc (0:56:2);
\draw[->,line width=1.4pt,color=darkred] (55:2) -- (61:2);
\node[color=darkred] at ($(60:2)+(0.3,0.15)$) {$x^2$};

\end{tikzpicture}
\end{document}
